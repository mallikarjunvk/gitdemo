pipeline {
    agent any

    stages {
        stage('find_user') {
            steps {
                sh '''
                    echo "finding the user"
                    whoami
                '''
            }
        }
            stage('find_working_dir') {
            steps {
                sh '''
                    echo "$pwd"
                    ls -ltr
                '''
            }
        }
        
    }
    
}
